tables- how to strucure the table

show tables;   - list of all tables

it create the structure
create table tablename (
    id int,
    name varchar(30),
    city varchar(30),
    mobile_no char(10)
)

to show the description of the table
desc tablename;

to  change the structure
to add field 
alter table tablename ADD fieldname datatype ;

to modify field datatype
alter table tablename MODIFY fieldname datatype;

to drop a columns
alter table tablename DROP COLUMN fieldname;

to rename a field 
alter table tablename RENAME COLUMN oldname TO newname;

to rename a table
alter table tablename RENAME TO  newtablename;

remove the table
drop table tname;

remove the values
truncate table tname;
